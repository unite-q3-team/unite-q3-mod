# Lines support key=value, comments start with # or //
# Key format: <voteName>.<attr>
# Attributes:
#   enabled:      1 or 0 (default 1)
#   action:       server command template; {arg} is replaced by the vote argument
#   usage:        message shown when argument is missing/invalid
#   requireArg:   1 if argument is required (default 0)
#   numericOnly:  1 if argument must be a number (default 0)

# Disable or enable built-ins
map.name = Map change
map.enabled = 1
map_restart.name = Map restart
map_restart.enabled = 1
map_restart.action = map_restart
nextmap.name = Next map
nextmap.enabled = 1
nextmap.action = rotate
rotate.enabled = 1

# Simple boolean toggles
instagib.name = Instagib
instagib.cvar = g_instagib
instagib.enabled = 1
instagib.action = g_instagib {arg}; map_restart
instagib.requireArg = 1
instagib.numericOnly = 1
instagib.usage = Usage: instagib <0|1>
